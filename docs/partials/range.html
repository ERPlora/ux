Componente Alpine.js

Uso del componente uxRange con controles programaticos.

Control programatico 
// Configuracion del componente uxRange
uxRange({
  value: 50,      // Valor inicial
  min: 0,         // Valor minimo
  max: 100,       // Valor maximo
  step: 5,        // Incremento
  disabled: false // Estado deshabilitado
})

// Metodos disponibles:
updateValue(newValue) // Establece un valor
increment(amount)     // Incrementa por step o cantidad
decrement(amount)     // Decrementa por step o cantidad
setMin()              // Establece al minimo
setMax()              // Establece al maximo
reset()               // Restaura al valor inicial

// Propiedades computadas:
percent      // Porcentaje actual (0-100)
thumbPosition // Posicion CSS del thumb

Componente uxDualRange

Para seleccionar rangos con dos valores.

// Configuracion del componente uxDualRange
uxDualRange({
  minValue: 20,  // Valor minimo seleccionado
  maxValue: 80,  // Valor maximo seleccionado
  min: 0,        // Limite inferior
  max: 100,      // Limite superior
  step: 1,       // Incremento
  minGap: 10,    // Separacion minima entre valores
  disabled: false
})

// Metodos:
updateMin(value) // Actualiza el valor minimo
updateMax(value) // Actualiza el valor maximo
reset()          // Restaura valores iniciales

// Propiedades computadas:
minPercent // Porcentaje del valor minimo
maxPercent // Porcentaje del valor maximo
fillStyle  // Estilos CSS para el fill { left, width }
range      // Array [minValue, maxValue]

Vista Django que filtra por rango

# views.py
from django.shortcuts import render
from django.http import HttpResponse
from .models import Product

def product_list(request):
    """
    Vista que filtra productos por rango de precio.
    Soporta peticiones HTMX para actualizacion parcial.
    """
    products = Product.objects.filter(is_active=True)

    # Filtrar por categoria
    category = request.GET.get('category')
    if category:
        products = products.filter(category__slug=category)

    # Filtrar por rango de precio
    price_min = request.GET.get('price_min')
    price_max = request.GET.get('price_max')

    if price_min:
        products = products.filter(price__gte=float(price_min))
    if price_max:
        products = products.filter(price__lte=float(price_max))

    # Ordenamiento
    sort = request.GET.get('sort', 'name')
    sort_options = {
        'price_asc': 'price',
        'price_desc': '-price',
        'name': 'name',
        'newest': '-created_at',
    }
    products = products.order_by(sort_options.get(sort, 'name'))

    context = {
        'products': products,
        'price_min': price_min or 0,
        'price_max': price_max or 1000,
        'total_count': products.count(),
    }

    # Si es peticion HTMX, devolver solo el partial
    if request.headers.get('HX-Request'):
        return render(request, 'products/partials/product_list.html', context)

    return render(request, 'products/list.html', context)

Configuracion de precios dinamica

Obtiene el rango de precios real desde la base de datos.

# views.py - Version avanzada con precios dinamicos
from django.db.models import Min, Max

def product_list_with_dynamic_range(request):
    """
    Vista con rango de precios calculado dinamicamente
    basado en los productos disponibles.
    """
    base_queryset = Product.objects.filter(is_active=True)

    # Obtener rango de precios real
    price_stats = base_queryset.aggregate(
        min_price=Min('price'),
        max_price=Max('price')
    )

    actual_min = int(price_stats['min_price'] or 0)
    actual_max = int(price_stats['max_price'] or 1000)

    # Valores seleccionados por el usuario
    price_min = request.GET.get('price_min', actual_min)
    price_max = request.GET.get('price_max', actual_max)

    # Filtrar productos
    products = base_queryset.filter(
        price__gte=price_min,
        price__lte=price_max
    )

    context = {
        'products': products,
        'price_min': price_min,
        'price_max': price_max,
        'actual_min': actual_min,
        'actual_max': actual_max,
    }

    if request.headers.get('HX-Request'):
        return render(request, 'products/partials/product_list.html', context)

    return render(request, 'products/list.html', context)
