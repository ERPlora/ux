Componente Alpine.js

// Configuracion del componente uxFab
uxFab({
  hideOnScroll: false,   // Ocultar al hacer scroll hacia abajo
  threshold: 50          // Umbral de scroll para ocultar (px)
})

// Metodos disponibles:
toggle()          // Alterna el speed dial
open()            // Abre el speed dial
close()           // Cierra el speed dial
handleScroll(y)   // Maneja el scroll para ocultar/mostrar
handleAction(fn)  // Ejecuta una accion y cierra

// Propiedades:
isOpen            // Estado del speed dial
hidden            // Si esta oculto por scroll

Vista Django para modal de creacion

# views.py
from django.shortcuts import render
from django.views.decorators.http import require_http_methods

@require_http_methods(["GET", "POST"])
def product_create(request):
    if request.method == "POST":
        form = ProductForm(request.POST)
        if form.is_valid():
            product = form.save()
            # Retorna el nuevo item para insertarlo en la lista
            return render(request, "products/partials/product_item.html", {
                "product": product
            })
    else:
        form = ProductForm()

    return render(request, "products/partials/create_modal.html", {
        "form": form
    })

Vista Django para contador del carrito

# views.py
from django.http import HttpResponse

def cart_count(request):
    """Retorna solo el numero de items en el carrito"""
    cart = request.session.get('cart', {})
    count = sum(item.get('quantity', 0) for item in cart.values())
    return HttpResponse(str(count) if count else "")
