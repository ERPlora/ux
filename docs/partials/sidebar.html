With React/Radix

The sidebar works with any framework - just apply the CSS classes.

function Sidebar({ collapsed, onToggle }) {
  return (
    <aside className="ux-sidebar" data-state={collapsed ? 'collapsed' : 'expanded'}>
      <div className="ux-sidebar__header">
        <a href="/" className="ux-sidebar__logo">
          <Logo />
          <span className="ux-sidebar__label">Acme</span>
        </a>
      </div>
      <div className="ux-sidebar__content">
        <nav className="ux-sidebar__menu">
          {menuItems.map(item => (
            <div key={item.id} className="ux-sidebar__menu-item">
              <NavLink to={item.href} className="ux-sidebar__menu-button">
                <span className="ux-sidebar__icon">{item.icon}</span>
                <span className="ux-sidebar__label">{item.label}</span>
              </NavLink>
            </div>
          ))}
        </nav>
      </div>
    </aside>
  );
}
