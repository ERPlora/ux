HTMX + Django

Integration examples with Django pagination for infinite scroll and load more patterns.

Django View

# views.py
from django.core.paginator import Paginator
from django.shortcuts import render

def product_list(request):
    products = Product.objects.all()
    paginator = Paginator(products, 10)  # 10 items per page
    page_number = request.GET.get('page', 1)
    page_obj = paginator.get_page(page_number)

    # Return partial template for HTMX requests
    if request.headers.get('HX-Request'):
        return render(request, 'partials/product_items.html', {
            'page_obj': page_obj,
        })

    return render(request, 'products/list.html', {
        'page_obj': page_obj,
    })

Filtered Results with Query Parameters

<!-- Preserve filters when loading more -->
<div class="ux-load-more">
  <button
    class="ux-load-more__button"
    hx-get="?page={{ page_obj.next_page_number }}&category={{ request.GET.category }}&sort={{ request.GET.sort }}"
    hx-target="#product-list"
    hx-swap="beforeend">
    Load More
  </button>
</div>

<!-- Or using Alpine.js to build URL dynamically -->
<div class="ux-load-more"
     x-data="{
       page: {{ page_obj.number }},
       hasNext: {{ page_obj.has_next|yesno:'true,false' }}
     }">
  <button
    x-show="hasNext"
    class="ux-load-more__button"
    :hx-get="'?page=' + (page + 1) + '&' + new URLSearchParams(window.location.search).toString()"
    hx-target="#product-list"
    hx-swap="beforeend"
    @htmx:afterRequest="page++; hasNext = $event.detail.xhr.getResponseHeader('X-Has-Next') === 'true'">
    Load More
  </button>
</div>

Django View with Custom Header

# views.py - Add custom header for pagination state
def product_list(request):
    products = Product.objects.all()
    paginator = Paginator(products, 10)
    page_number = request.GET.get('page', 1)
    page_obj = paginator.get_page(page_number)

    if request.headers.get('HX-Request'):
        response = render(request, 'partials/product_items.html', {
            'page_obj': page_obj,
        })
        # Add header for client-side state management
        response['X-Has-Next'] = str(page_obj.has_next()).lower()
        response['X-Current-Page'] = str(page_obj.number)
        response['X-Total-Pages'] = str(paginator.num_pages)
        return response

    return render(request, 'products/list.html', {'page_obj': page_obj})
