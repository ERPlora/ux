HTMX + Django

Actualizacion de stock en tiempo real con HTMX y Django.

Template Django

<!-- templates/components/stock_indicator.html -->
<span id="stock-{{ product.id }}"
      class="ux-stock-indicator ux-stock-indicator--badge {{ stock_class }}"
      hx-get="{% url 'stock_status' product.id %}"
      hx-trigger="every 30s"
      hx-swap="outerHTML">
  <span class="ux-stock-indicator__dot"></span>
  <span class="ux-stock-indicator__count">{{ product.stock }}</span>
  <span class="ux-stock-indicator__label">{{ stock_label }}</span>
</span>

Vista Django

# views.py
from django.shortcuts import render, get_object_or_404
from .models import Product

def stock_status(request, product_id):
    product = get_object_or_404(Product, id=product_id)

    # Determinar estado y clase CSS
    if product.stock == 0:
        stock_class = 'ux-stock-indicator--out-of-stock'
        stock_label = 'Out of Stock'
    elif product.stock <= product.low_stock_threshold:
        stock_class = 'ux-stock-indicator--low-stock'
        stock_label = 'Low Stock'
    else:
        stock_class = 'ux-stock-indicator--in-stock'
        stock_label = 'In Stock'

    return render(request, 'components/stock_indicator.html', {
        'product': product,
        'stock_class': stock_class,
        'stock_label': stock_label,
    })

Vista Add to Cart

# views.py
from django.db import transaction

def add_to_cart(request, product_id):
    product = get_object_or_404(Product, id=product_id)

    with transaction.atomic():
        # Decrementar stock
        product.stock = max(0, product.stock - 1)
        product.save()

        # Agregar al carrito del usuario
        cart, _ = Cart.objects.get_or_create(user=request.user)
        CartItem.objects.update_or_create(
            cart=cart,
            product=product,
            defaults={'quantity': F('quantity') + 1}
        )

    # Retornar el indicador actualizado
    return stock_status(request, product_id)

# consumers.py
import json
from channels.generic.websocket import AsyncWebsocketConsumer

class StockConsumer(AsyncWebsocketConsumer):
    async def connect(self):
        await self.channel_layer.group_add('stock_updates', self.channel_name)
        await self.accept()

    async def stock_update(self, event):
        # Enviar HTML del indicador actualizado
        await self.send(text_data=event['html'])
