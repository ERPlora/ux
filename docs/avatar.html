Helpers JavaScript

// Obtener iniciales de un nombre
UX.getInitials('Juan Perez')  // "JP"
UX.getInitials('Ana Maria Lopez', 3)  // "AML"

// Generar color consistente desde string
UX.stringToColor('Juan Perez')  // "primary" | "secondary" | etc.

Integracion con HTMX

Avatares dinamicos cargados desde el servidor.

<!-- Avatar que carga imagen del servidor -->
<div class="ux-avatar ux-avatar--lg ux-avatar--loading"
     hx-get="/api/user/123/avatar"
     hx-trigger="load"
     hx-swap="outerHTML">
</div>

<!-- Respuesta del servidor -->
<div class="ux-avatar ux-avatar--lg">
  <img src="/uploads/avatar-123.jpg" alt="Usuario">
  <span class="ux-avatar__status ux-avatar__status--online"></span>
</div>
<!-- Avatar clickable que abre perfil -->
<div class="ux-avatar ux-avatar--lg ux-avatar--clickable"
     hx-get="/api/user/123/profile"
     hx-target="#user-profile-modal"
     hx-trigger="click">
  <img src="/avatar.jpg" alt="">
</div>

<div id="user-profile-modal"></div>
<!-- Grupo de avatares cargado dinamicamente -->
<div class="ux-avatar-group"
     hx-get="/api/project/456/members"
     hx-trigger="load">
  <!-- Loading placeholders -->
  <div class="ux-avatar ux-avatar--loading"></div>
  <div class="ux-avatar ux-avatar--loading"></div>
  <div class="ux-avatar ux-avatar--loading"></div>
</div>

<!-- Respuesta del servidor -->
<div class="ux-avatar"><img src="/avatar1.jpg"></div>
<div class="ux-avatar"><img src="/avatar2.jpg"></div>
<div class="ux-avatar"><img src="/avatar3.jpg"></div>
<div class="ux-avatar-group__more">+5</div>
<!-- Actualizar estado del usuario en tiempo real -->
<div class="ux-avatar ux-avatar--lg"
     hx-get="/api/user/123/status"
     hx-trigger="every 30s"
     hx-swap="innerHTML">
  <img src="/avatar.jpg" alt="">
  <span class="ux-avatar__status ux-avatar__status--online"></span>
</div>
