Modales que aparecen desde los lados. Usa .ux-modal-backdrop--side + .ux-modal-backdrop--left/right en el backdrop y .ux-modal--left/right en el modal.

  <div class="ux-modal ux-modal--left" style="width: 320px;">
    ...
  </div>
</div>

<!-- Modal lateral derecha -->

  <div class="ux-modal ux-modal--right" style="width: 400px;">
    ...
  </div>
</div>

Sin Cierre al Click en Backdrop

Usa closeOnBackdrop: false para deshabilitar el cierre al hacer click fuera del modal.

<!-- Deshabilitar cierre en backdrop -->
<div x-data="uxModal({ closeOnBackdrop: false })">
  ...
</div>

Integracion con HTMX

Ejemplos de como usar modales con HTMX para cargar contenido dinamico desde el servidor.

Cargar contenido en modal

Click en el boton para cargar detalles del usuario desde el servidor.

<!-- Boton que abre modal y carga contenido -->
<button class="ux-button ux-color-primary"
        hx-get="/api/users/123"
        hx-target="#modal-content"
        hx-trigger="click"
        onclick="document.getElementById('user-modal').checked = true">
  Ver Usuario
</button>

<!-- Modal CSS-only con contenido dinamico -->
<input type="checkbox" id="user-modal" class="ux-modal-toggle" hidden>
<div class="ux-modal-backdrop">
  <div class="ux-modal">
    <div class="ux-modal__header">
      <h2 class="ux-modal__title">Detalles de Usuario</h2>
      <label for="user-modal" class="ux-modal__close">×</label>
    </div>
    <div id="modal-content" class="ux-modal__content">
      <!-- Contenido cargado via HTMX -->
      <div class="ux-skeleton"></div>
    </div>
  </div>
</div>

Formulario en modal con HTMX

Formulario que envia datos sin recargar y cierra el modal al exito.

<!-- Formulario con HTMX en modal -->
<form hx-post="/api/contacts"
      hx-target="#contacts-list"
      hx-swap="beforeend"
      hx-on::after-request="if(event.detail.successful) document.getElementById('contact-modal').checked = false">

  <div class="ux-input-group">
    <label class="ux-input__label">Nombre</label>
    <input type="text" name="name" class="ux-input" required>
  </div>

  <div class="ux-input-group">
    <label class="ux-input__label">Email</label>
    <input type="email" name="email" class="ux-input" required>
  </div>

  <div class="ux-modal__footer">
    <label for="contact-modal" class="ux-button">Cancelar</label>
    <button type="submit" class="ux-button ux-color-primary">
      <span class="htmx-indicator ux-spinner ux-spinner--sm"></span>
      Guardar
    </button>
  </div>
</form>

Confirmacion de eliminacion

Modal de confirmacion antes de eliminar un elemento.

<!-- Boton que abre modal de confirmacion -->
<button class="ux-button ux-color-danger--outline"
        onclick="document.getElementById('delete-modal').checked = true;
                 document.getElementById('delete-id').value = 123">
  Eliminar
</button>

<!-- Modal de confirmacion -->
<input type="checkbox" id="delete-modal" class="ux-modal-toggle" hidden>
<div class="ux-modal-backdrop">
  <div class="ux-modal">
    <div class="ux-modal__content ux-modal__content--centered">
      <h3>Confirmar eliminacion</h3>
      <p>Esta accion no se puede deshacer.</p>

      <input type="hidden" id="delete-id" name="id">

      <div class="ux-modal__footer">
        <label for="delete-modal" class="ux-button">Cancelar</label>
        <button class="ux-button ux-color-danger"
                hx-delete="/api/items"
                hx-include="#delete-id"
                hx-target="closest .item-row"
                hx-swap="outerHTML"
                hx-on::after-request="document.getElementById('delete-modal').checked = false">
          Eliminar
        </button>
      </div>
    </div>
  </div>
</div>

Editar en modal

Cargar formulario de edicion y actualizar la fila al guardar.

<!-- Fila de tabla con boton editar -->
<tr id="product-row-123">
  <td>Producto A</td>
  <td>$99.00</td>
  <td>
    <button class="ux-button ux-button--sm"
            hx-get="/api/products/123/edit"
            hx-target="#edit-form-container"
            onclick="document.getElementById('edit-modal').checked = true">
      Editar
    </button>
  </td>
</tr>

<!-- Modal de edicion -->
<input type="checkbox" id="edit-modal" class="ux-modal-toggle" hidden>
<div class="ux-modal-backdrop">
  <div class="ux-modal">
    <div class="ux-modal__header">
      <h2 class="ux-modal__title">Editar Producto</h2>
      <label for="edit-modal" class="ux-modal__close">×</label>
    </div>
    <div id="edit-form-container" class="ux-modal__content">
      <!-- Servidor devuelve formulario pre-llenado -->
    </div>
  </div>
</div>

<!-- Respuesta del servidor: formulario de edicion -->
<form hx-put="/api/products/123"
      hx-target="#product-row-123"
      hx-swap="outerHTML"
      hx-on::after-request="document.getElementById('edit-modal').checked = false">
  <div class="ux-input-group">
    <label class="ux-input__label">Nombre</label>
    <input type="text" name="name" value="Producto A" class="ux-input">
  </div>
  <div class="ux-input-group">
    <label class="ux-input__label">Precio</label>
    <input type="number" name="price" value="99" class="ux-input">
  </div>
  <button type="submit" class="ux-button ux-color-primary">Guardar</button>
</form>

Wizard multi-paso en modal

Modal con pasos que se cargan dinamicamente desde el servidor.

<!-- Modal wizard -->
<div class="ux-modal">
  <div class="ux-modal__header">
    <h2 class="ux-modal__title">Crear Cuenta</h2>
  </div>

  <div id="wizard-step" class="ux-modal__content">
    <!-- Paso 1: Datos basicos -->
    <form hx-post="/api/signup/step1"
          hx-target="#wizard-step"
          hx-swap="innerHTML">
      <div class="ux-input-group">
        <label>Email</label>
        <input type="email" name="email" class="ux-input" required>
      </div>
      <button type="submit" class="ux-button ux-color-primary">
        Siguiente
      </button>
    </form>
  </div>
</div>

<!-- Servidor devuelve Paso 2 -->
<form hx-post="/api/signup/step2"
      hx-target="#wizard-step"
      hx-swap="innerHTML">
  <div class="ux-input-group">
    <label>Password</label>
    <input type="password" name="password" class="ux-input">
  </div>
  <button type="submit" class="ux-button ux-color-primary">
    Crear Cuenta
  </button>
</form>

API Alpine

Config / Metodo Descripcion
closeOnBackdrop Boolean (default: true) - Cerrar al click en backdrop
closeOnEscape Boolean (default: true) - Cerrar con tecla Escape
isOpen Boolean - Estado del modal
open() Abre el modal
close() Cierra el modal
toggle() Alterna el estado

Clases CSS

Clase Descripcion
.ux-modal-backdrop Contenedor overlay fijo
.ux-modal-backdrop--open Estado visible
.ux-modal El modal en si
.ux-modal--full-height Modal altura completa
.ux-modal-backdrop--side Backdrop para modales laterales
.ux-modal-backdrop--left Alinea modal a la izquierda
.ux-modal-backdrop--right Alinea modal a la derecha
.ux-modal--left Modal lateral izquierda (slide-in)
.ux-modal--right Modal lateral derecha (slide-in)
.ux-modal--glass Variante glass (iOS 26)
.ux-modal__handle Handle arrastrable (solo movil)
.ux-modal__header Cabecera con titulo y close
.ux-modal__content Contenido scrollable
.ux-modal__content--centered Contenido centrado
.ux-modal__footer Pie con botones
