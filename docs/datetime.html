Componente Alpine.js

// Configuracion del componente uxDatetime
uxDatetime({
  value: null,           // Valor inicial (ISO string)
  mode: 'date',          // 'date', 'time', 'datetime'
  format: 'YYYY-MM-DD',  // Formato de salida
  min: null,             // Fecha minima
  max: null,             // Fecha maxima
  use24Hour: false       // Usar formato 24 horas
})

// Propiedades:
isOpen           // Si el picker esta abierto
value            // Valor seleccionado (ISO string)
displayValue     // Valor formateado para mostrar
view             // Vista actual ('days', 'months', 'years')
currentMonth     // Mes actual (0-11)
currentYear      // Ano actual
hours            // Hora seleccionada
minutes          // Minutos seleccionados
period           // 'AM' o 'PM'
daysInMonth      // Array de dias del mes actual
weekdays         // Array de nombres de dias
months           // Array de nombres de meses

// Metodos:
open()           // Abre el picker
close()          // Cierra el picker
toggle()         // Alterna abrir/cerrar
prevMonth()      // Mes anterior
nextMonth()      // Mes siguiente
selectDate(date) // Selecciona una fecha
selectMonth(m)   // Selecciona un mes
selectYear(y)    // Selecciona un ano
isToday(date)    // Verifica si es hoy
isSelected(date) // Verifica si esta seleccionado
confirm()        // Confirma la seleccion
clear()          // Limpia la seleccion

Vista Django para verificar disponibilidad

# views.py
from django.http import HttpResponse
from django.views.decorators.http import require_POST
from datetime import datetime

@require_POST
def check_availability(request):
    date_str = request.POST.get('booking_date')
    if not date_str:
        return HttpResponse('<span class="ux-text-secondary">Selecciona una fecha</span>')

    booking_date = datetime.strptime(date_str, '%Y-%m-%d').date()

    # Verificar disponibilidad en la base de datos
    is_available = not Booking.objects.filter(
        date=booking_date,
        status='confirmed'
    ).exists()

    if is_available:
        return HttpResponse(
            '<span class="ux-badge ux-color-success--soft">'
            'Fecha disponible</span>'
        )
    return HttpResponse(
        '<span class="ux-badge ux-color-danger--soft">'
        'Fecha no disponible</span>'
    )

Vista Django para filtrar por rango

# views.py
from django.shortcuts import render
from datetime import datetime

def report_list(request):
    reports = Report.objects.all()

    date_from = request.GET.get('date_from')
    date_to = request.GET.get('date_to')

    if date_from:
        reports = reports.filter(
            created_at__date__gte=datetime.strptime(date_from, '%Y-%m-%d').date()
        )

    if date_to:
        reports = reports.filter(
            created_at__date__lte=datetime.strptime(date_to, '%Y-%m-%d').date()
        )

    context = {
        'reports': reports,
        'filter': {'date_from': date_from, 'date_to': date_to}
    }

    # Responder con partial si es HTMX
    if request.headers.get('HX-Request'):
        return render(request, 'reports/partials/results.html', context)

    return render(request, 'reports/list.html', context)

Modelo y Form Django

# models.py
from django.db import models

class Event(models.Model):
    title = models.CharField(max_length=200)
    event_datetime = models.DateTimeField()
    registration_deadline = models.DateField(null=True, blank=True)
    created_at = models.DateTimeField(auto_now_add=True)

# forms.py
from django import forms
from .models import Event

class EventForm(forms.ModelForm):
    class Meta:
        model = Event
        fields = ['title', 'event_datetime', 'registration_deadline']
        widgets = {
            'event_datetime': forms.DateTimeInput(
                attrs={'type': 'datetime-local', 'class': 'ux-datetime__native'}
            ),
            'registration_deadline': forms.DateInput(
                attrs={'type': 'date', 'class': 'ux-datetime__native'}
            ),
        }
