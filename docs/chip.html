Componentes Alpine.js

// uxChip - Chip individual seleccionable
uxChip() => {
  selected: false,
  toggle()   // Alterna estado
  select()   // Selecciona
  deselect() // Deselecciona
}

// uxChipGroup - Grupo de chips
uxChipGroup({
  multiple: false  // true para seleccion multiple
}) => {
  selected: null | [],     // Valor(es) seleccionado(s)
  isSelected(value)        // Verifica si esta seleccionado
  toggle(value)            // Alterna seleccion
  select(value)            // Selecciona un valor
  clear()                  // Limpia la seleccion
}

Vista Django

# views.py
from django.shortcuts import render

def productos_lista(request):
    categoria = request.GET.get('categoria', 'todos')
    categorias = Categoria.objects.all()

    if categoria == 'todos':
        productos = Producto.objects.all()
    else:
        productos = Producto.objects.filter(categoria__slug=categoria)

    context = {
        'productos': productos,
        'categorias': categorias,
        'categoria_activa': categoria,
    }

    # Si es HTMX, solo el grid
    if request.headers.get('HX-Request'):
        return render(request, 'productos/partials/grid.html', context)

    return render(request, 'productos/lista.html', context)
