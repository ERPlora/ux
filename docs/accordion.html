Componente Alpine.js

// Configuracion del componente uxAccordion
uxAccordion({
  openItems: [0],     // Indices de items abiertos inicialmente
  multiple: false,    // Permitir multiples abiertos
  disabled: false     // Deshabilitar todo el acordeon
})

// Metodos disponibles:
isOpen(index)         // Verifica si un item esta abierto
toggle(index)         // Alterna un item
open(index)           // Abre un item
close(index)          // Cierra un item
openAll(total)        // Abre todos los items
closeAll()            // Cierra todos los items

// Componente individual uxAccordionItem
uxAccordionItem({
  isOpen: false,      // Estado inicial
  disabled: false     // Deshabilitar item
})

Integracion con HTMX

Acordeones con contenido dinamico desde el servidor.

<!-- Cargar contenido al abrir -->
<div class="ux-accordion" x-data="uxAccordion()">
  <div class="ux-accordion-item" :class="{ 'ux-accordion-item--open': isOpen(0) }">
    <button class="ux-accordion-item__header"
            @click="toggle(0)"
            hx-get="/api/faq/1"
            hx-target="#faq-content-1"
            hx-trigger="click once">
      <span class="ux-accordion-item__title">Pregunta 1</span>
      <span class="ux-accordion-item__chevron">...</span>
    </button>
    <div class="ux-accordion-item__body">
      <div class="ux-accordion-item__body-inner">
        <div class="ux-accordion-item__body-content" id="faq-content-1">
          <span class="ux-spinner"></span> Cargando...
        </div>
      </div>
    </div>
  </div>
</div>
<!-- Cargar lista completa de FAQ desde servidor -->
<div class="ux-accordion ux-accordion--faq"
     hx-get="/api/faqs"
     hx-trigger="load">
  <!-- Skeleton loading -->
  <div class="ux-skeleton" style="height: 48px;"></div>
  <div class="ux-skeleton" style="height: 48px;"></div>
  <div class="ux-skeleton" style="height: 48px;"></div>
</div>

<!-- Respuesta del servidor -->
<div class="ux-accordion-item">
  <button class="ux-accordion-item__header" ...>
    <span class="ux-accordion-item__title">Como funciona?</span>
  </button>
  ...
</div>
<!-- Buscar en FAQ -->
<input type="search"
       class="ux-input"
       placeholder="Buscar pregunta..."
       hx-get="/api/faqs/search"
       hx-trigger="input changed delay:300ms"
       hx-target="#faq-accordion">

<div id="faq-accordion" class="ux-accordion ux-accordion--faq">
  <!-- Resultados filtrados -->
</div>
