uxProductionLine({
  name: 'Production Line',    // Line name
  status: 'running',          // 'running' | 'stopped' | 'maintenance' | 'error'
  target: 1000,               // Daily production target
  actual: 0,                  // Current production count
  efficiency: 0,              // Efficiency percentage (0-100)
  speed: 0,                   // Current speed (0-100)
  maxSpeed: 100,              // Maximum speed
  downtime: 0,                // Downtime in minutes
  showFlow: true,             // Show animated flow in connectors
  lastUpdate: null,           // Last update timestamp
  updateInterval: null,       // Auto-refresh interval in ms (dispatches event)
  stations: [                 // Array of station objects
    {
      name: 'Station 1',      // Station name
      status: 'running',      // Station status
      output: 500,            // Station output count
      alerts: 0,              // Number of alerts
      icon: 'custom'          // Optional custom icon key
    }
  ]
})

// Methods
setStatus(status)                 // Set overall line status
updateActual(value)               // Update production count
updateStation(index, data)        // Update station data
setStationStatus(index, status)   // Set station status
addAlert(stationIndex, count)     // Add alerts to station
clearAlerts(stationIndex)         // Clear station alerts
updateMetrics(data)               // Batch update metrics

// Computed Properties
progressPercent                   // Target progress (0-100)
progressStatus                    // 'success' | 'primary' | 'warning' | 'danger'
speedPercent                      // Speed percentage
efficiencyDisplay                 // Formatted efficiency
downtimeDisplay                   // Formatted downtime (12m or 2h 30m)
isRunning                         // Boolean: is line running
runningStationsCount              // Count of running stations
errorStationsCount                // Count of stations in error

// Events
@productionline:statuschange      // { status }
@productionline:update            // { actual, progress }
@productionline:stationupdate     // { index, station }
@productionline:stationstatus     // { index, status }
@productionline:metricsupdate     // { ...metrics }
@productionline:refresh           // Auto-refresh trigger
