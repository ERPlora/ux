# views.py from django.shortcuts import render from django.http import HttpResponse def team_members(request): members = TeamMember.objects.filter(team=request.user.team) return render(request, 'team/partials/members_list.html', { 'members': members }) def delete_member(request, member_id): if request.method == 'DELETE': member = get_object_or_404(TeamMember, id=member_id) member.delete() return HttpResponse('') # Empty response removes the element return HttpResponse(status=405)
