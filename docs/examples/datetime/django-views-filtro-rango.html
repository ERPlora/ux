# views.py from django.shortcuts import render from datetime import datetime def report_list(request): reports = Report.objects.all() date_from = request.GET.get('date_from') date_to = request.GET.get('date_to') if date_from: reports = reports.filter( created_at__date__gte=datetime.strptime(date_from, '%Y-%m-%d').date() ) if date_to: reports = reports.filter( created_at__date__lte=datetime.strptime(date_to, '%Y-%m-%d').date() ) context = { 'reports': reports, 'filter': {'date_from': date_from, 'date_to': date_to} } # Responder con partial si es HTMX if request.headers.get('HX-Request'): return render(request, 'reports/partials/results.html', context) return render(request, 'reports/list.html', context)
