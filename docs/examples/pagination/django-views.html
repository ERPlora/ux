# views.py from django.core.paginator import Paginator from django.shortcuts import render def product_list(request): products = Product.objects.all() per_page = int(request.GET.get('per_page', 10)) paginator = Paginator(products, per_page) page_number = request.GET.get('page', 1) page_obj = paginator.get_page(page_number) template = 'products/_list.html' if request.htmx else 'products/list.html' return render(request, template, { 'page_obj': page_obj, 'per_page': per_page, })
