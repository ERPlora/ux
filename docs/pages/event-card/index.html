Integracion con HTMX

Ejemplos de como usar Event Cards con HTMX para cargar detalles o gestionar RSVPs.

<!-- Click para cargar detalles -->
<div class="ux-event-card ux-event-card--clickable"
     hx-get="/events/123/"
     hx-target="#event-modal .ux-modal__content"
     hx-trigger="click">
  ...
</div>

<!-- RSVP con HTMX -->
<div class="ux-event-card__footer">
  <button class="ux-button ux-button--sm ux-color-success"
          hx-post="/events/123/rsvp/"
          hx-vals='{"status": "going"}'
          hx-target="closest .ux-event-card__footer-start"
          hx-swap="innerHTML">
    Asistire
  </button>
</div>

<!-- Lista de eventos con paginacion -->
<div id="events-list" class="ux-event-card-list">
  <!-- Events loaded here -->
</div>
<div hx-get="/events/?page=2"
     hx-trigger="revealed"
     hx-target="#events-list"
     hx-swap="beforeend">
  <div class="ux-spinner"></div>
</div>

Ejemplo con Django

Template partial para renderizar eventos desde Django.

<!-- partials/event_card.html -->
<div class="ux-event-card ux-event-card--{{ event.category }} {% if event.is_all_day %}ux-event-card--all-day{% endif %} {% if event.is_recurring %}ux-event-card--recurring{% endif %} {% if event.is_past %}ux-event-card--past{% endif %}"
     hx-get="{% url 'event_detail' event.id %}"
     hx-target="#event-modal-content"
     hx-trigger="click">
  <div class="ux-event-card__header">
    <div class="ux-event-card__header-content">
      <h4 class="ux-event-card__title">{{ event.title }}</h4>
      {% if event.description %}
      <p class="ux-event-card__description">{{ event.description|truncatewords:15 }}</p>
      {% endif %}
    </div>
    <span class="ux-event-card__category">{{ event.get_category_display }}</span>
  </div>
  <div class="ux-event-card__content">
    <div class="ux-event-card__meta">
      {% if event.is_all_day %}
      <span class="ux-event-card__all-day-badge">Todo el dia</span>
      {% else %}
      <span class="ux-event-card__time">
        <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
          <circle cx="12" cy="12" r="10"/><path d="M12 6v6l4 2"/>
        </svg>
        {{ event.start_time|time:"g:i A" }} - {{ event.end_time|time:"g:i A" }}
      </span>
      {% endif %}
    </div>
    {% if event.location %}
    <div class="ux-event-card__location">
      <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">
        <path d="M21 10c0 7-9 13-9 13s-9-6-9-13a9 9 0 0118 0z"/>
        <circle cx="12" cy="10" r="3"/>
      </svg>
      <span>{{ event.location }}</span>
    </div>
    {% endif %}
    {% if event.attendees.exists %}
    <div class="ux-event-card__attendees">
      <div class="ux-event-card__attendees-stack">
        {% for attendee in event.attendees.all|slice:":3" %}
        <span class="ux-avatar">{{ attendee.initials }}</span>
        {% endfor %}
        {% if event.attendees.count > 3 %}
        <span class="ux-event-card__attendees-more">+{{ event.attendees.count|add:"-3" }}</span>
        {% endif %}
      </div>
    </div>
    {% endif %}
  </div>
</div>
# views.py
from django.shortcuts import render
from .models import Event

def calendar_view(request):
    events = Event.objects.filter(
        user=request.user,
        start_date__gte=timezone.now().date()
    ).select_related('category').prefetch_related('attendees')

    return render(request, 'calendar.html', {'events': events})

def event_detail(request, event_id):
    event = get_object_or_404(Event, id=event_id, user=request.user)
    return render(request, 'partials/event_detail.html', {'event': event})
