# views.py
from django.http import HttpResponse
from django.shortcuts import render

class CheckoutWizardView:
    STEPS = {
        1: 'checkout/partials/step_shipping.html',
        2: 'checkout/partials/step_payment.html',
        3: 'checkout/partials/step_confirm.html',
    }

    def shipping(self, request):
        if request.method == 'POST':
            form = ShippingForm(request.POST)
            if form.is_valid():
                request.session['shipping'] = form.cleaned_data
                response = render(request, self.STEPS[2], {'form': PaymentForm()})
                response['X-Step'] = '2'
                return response
            return render(request, self.STEPS[1], {'form': form})

        form = ShippingForm(initial=request.session.get('shipping', {}))
        response = render(request, self.STEPS[1], {'form': form})
        response['X-Step'] = '1'
        return response

    def payment(self, request):
        if request.method == 'POST':
            form = PaymentForm(request.POST)
            if form.is_valid():
                request.session['payment'] = form.cleaned_data
                # Prepare order summary
                context = {
                    'shipping': request.session['shipping'],
                    'payment': form.cleaned_data,
                    'cart': get_cart(request),
                }
                response = render(request, self.STEPS[3], context)
                response['X-Step'] = '3'
                return response
            return render(request, self.STEPS[2], {'form': form})

        form = PaymentForm(initial=request.session.get('payment', {}))
        response = render(request, self.STEPS[2], {'form': form})
        response['X-Step'] = '2'
        return response

    def confirm(self, request):
        if request.method == 'POST':
            # Process the order
            order = Order.objects.create(
                shipping_address=request.session['shipping'],
                payment_method=request.session['payment'],
                user=request.user,
            )
            # Clear session
            del request.session['shipping']
            del request.session['payment']
            # Redirect to success
            return HttpResponse(
                headers={'HX-Redirect': f'/orders/{order.id}/success/'}
            )

        context = {
            'shipping': request.session['shipping'],
            'payment': request.session['payment'],
            'cart': get_cart(request),
        }
        response = render(request, self.STEPS[3], context)
        response['X-Step'] = '3'
        return response

# views.py - Onboarding with step validation
from django.views import View

class OnboardingView(View):
    STEPS = {
        1: ('onboarding/partials/account.html', AccountForm),
        2: ('onboarding/partials/company.html', CompanyForm),
        3: ('onboarding/partials/team.html', TeamForm),
    }

    def get_step(self, request, step):
        template, form_class = self.STEPS[step]
        form = form_class(initial=request.session.get(f'step_{step}', {}))
        response = render(request, template, {'form': form, 'step': step})
        response['X-Step'] = str(step)
        return response

    def post_step(self, request, step):
        template, form_class = self.STEPS[step]
        form = form_class(request.POST)

        if form.is_valid():
            # Save step data to session
            request.session[f'step_{step}'] = form.cleaned_data

            if step < 3:
                # Move to next step
                next_template, next_form_class = self.STEPS[step + 1]
                next_form = next_form_class()
                response = render(request, next_template, {
                    'form': next_form,
                    'step': step + 1
                })
                response['X-Step'] = str(step + 1)
                return response
            else:
                # Complete onboarding
                self.complete_onboarding(request)
                return HttpResponse(headers={'HX-Redirect': '/dashboard/'})

        # Validation failed, return form with errors
        response = render(request, template, {'form': form, 'step': step})
        response['X-Step'] = str(step)
        return response

    def complete_onboarding(self, request):
        # Create company and user profile
        company = Company.objects.create(**request.session['step_2'])
        request.user.profile.company = company
        request.user.profile.onboarding_complete = True
        request.user.profile.save()

        # Send team invitations
        for email in request.session['step_3'].get('emails', []):
            send_invitation(company, email)

        # Clear session data
        for i in range(1, 4):
            request.session.pop(f'step_{i}', None)
