Alpine.js Options

uxPerformanceMeter({
  // Employee data
  employee: {
    name: 'Employee Name',
    role: 'Role',
    department: 'Department',
    avatar: null,      // Image URL
    initials: null     // Override auto-generated initials
  },

  // Score configuration
  score: 0,            // Current score value
  maxScore: 100,       // Maximum score (for percentage/points)
  scoreType: 'percentage', // 'percentage' | 'points' | 'rating'
  ratingMax: 5,        // Max rating (when scoreType is 'rating')

  // Goal tracking
  goal: null,          // Target goal value
  actual: null,        // Actual achieved value

  // Categories breakdown
  categories: [
    { name: 'Category', score: 85, trend: 5 }
  ],

  // Trend indicator
  trend: null,         // Change from previous period (+/- number)
  trendLabel: 'vs last period',

  // Period
  period: 'quarterly', // Current period
  periods: ['monthly', 'quarterly', 'yearly'],

  // Animation
  animated: true,      // Enable fill animations

  // Thresholds for color coding
  thresholds: {
    excellent: 90,     // >= 90 = green
    good: 70,          // >= 70 = blue
    average: 50        // >= 50 = yellow, < 50 = red
  }
})

// Computed properties
scoreLevel          // 'excellent' | 'good' | 'average' | 'poor'
scoreLevelLabel     // Human-readable label
normalizedScore     // Score as 0-100 percentage
displayScore        // Formatted score for display
trendDirection      // 'up' | 'down' | 'neutral'
goalDifference      // actual - goal

// Methods
setPeriod(period)   // Change period, dispatches event
updateScore(value)  // Update score with event
getInitials()       // Get employee initials
getCategoryLevel(score)  // Get level for a score

// Events
@performance-meter:period-change  // { period }
@performance-meter:score-change   // { score, level }

Django View

# views.py
from django.shortcuts import render, get_object_or_404
from django.http import JsonResponse
import json

def employee_performance(request, employee_id):
    employee = get_object_or_404(Employee, id=employee_id)
    period = request.GET.get('period', 'quarterly')

    # Get performance data for the period
    performance = PerformanceReview.objects.filter(
        employee=employee,
        period=period
    ).latest('created_at')

    categories = [
        {'name': c.name, 'score': c.score, 'trend': c.trend}
        for c in performance.categories.all()
    ]

    context = {
        'employee': employee,
        'performance': {
            'overall_score': performance.overall_score,
            'score_type': 'percentage',
            'trend': performance.trend_vs_previous,
            'goal': performance.goal,
            'actual': performance.overall_score,
            'period': period,
            'categories': json.dumps(categories)
        }
    }

    return render(request, 'partials/employee_performance.html', context)
