uxQualityCheck({
  // Configuracion
  title: 'Quality Check',         // Titulo del formulario
  subtitle: '',                   // Subtitulo opcional
  orderNumber: '',                // Numero de orden
  productName: '',                // Nombre del producto
  batchNumber: '',                // Numero de lote
  inspector: '',                  // Nombre del inspector
  defectCategories: [...],        // Categorias de defectos

  // Items del checklist
  checklistItems: [
    {
      id: 'unique-id',
      name: 'Item Name',
      description: 'Optional description',
      required: true,
      status: null,               // 'pass', 'fail', 'na', null

      // Para mediciones
      hasMeasurement: true,
      measurementUnit: 'mm',
      measurementMin: 10.0,
      measurementMax: 10.5,
      measurementTarget: 10.25
    }
  ]
})

// Metodos
setStatus(itemId, status)         // Cambiar estado del item
updateMeasurement(itemId, value)  // Actualizar medicion
addPhoto(itemId, file)            // Agregar foto
removePhoto(itemId, photoId)      // Eliminar foto
toggleDefect(itemId, defect)      // Toggle categoria de defecto
complete()                        // Finalizar inspeccion
print()                           // Imprimir formulario
getData()                         // Obtener datos para envio

// Propiedades computadas
passCount                         // Cantidad de items aprobados
failCount                         // Cantidad de items rechazados
naCount                           // Cantidad de items N/A
pendingCount                      // Cantidad de items pendientes
completionPercent                 // Porcentaje completado
overallStatus                     // 'pending', 'pass', 'fail'
isComplete                        // Boolean si requeridos estan completos

// Eventos
@qualitycheck:statuschange        // Al cambiar estado de item
@qualitycheck:measurementchange   // Al cambiar medicion
@qualitycheck:photochange         // Al agregar/eliminar foto
@qualitycheck:defectchange        // Al cambiar defectos
@qualitycheck:complete            // Al completar inspeccion
@qualitycheck:incomplete          // Si hay items requeridos pendientes
