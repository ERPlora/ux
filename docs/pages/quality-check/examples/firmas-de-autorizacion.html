# models.py
from django.db import models
from django.contrib.postgres.fields import ArrayField

class Inspection(models.Model):
    STATUS_CHOICES = [
        ('pending', 'Pending'),
        ('pass', 'Pass'),
        ('fail', 'Fail'),
    ]

    order = models.ForeignKey('Order', on_delete=models.CASCADE)
    product = models.ForeignKey('Product', on_delete=models.CASCADE)
    inspector = models.ForeignKey('User', on_delete=models.SET_NULL, null=True)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='pending')
    started_at = models.DateTimeField(auto_now_add=True)
    completed_at = models.DateTimeField(null=True, blank=True)
    inspector_signature = models.TextField(blank=True)
    supervisor_signature = models.TextField(blank=True)


class InspectionItem(models.Model):
    inspection = models.ForeignKey(Inspection, on_delete=models.CASCADE, related_name='items')
    item_id = models.CharField(max_length=100)
    status = models.CharField(max_length=20, null=True)
    measurement_value = models.DecimalField(max_digits=10, decimal_places=4, null=True)
    notes = models.TextField(blank=True)
    defects = ArrayField(models.CharField(max_length=100), default=list)


class InspectionPhoto(models.Model):
    inspection_item = models.ForeignKey(InspectionItem, on_delete=models.CASCADE, related_name='photos')
    image = models.ImageField(upload_to='inspection_photos/')
    uploaded_at = models.DateTimeField(auto_now_add=True)
