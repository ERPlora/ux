  initialExpanded: 2,      // Profundidad de expansion inicial

expandAll()               // Expandir todos los nodos
collapseAll()             // Colapsar todos los nodos

setData(data)             // Actualizar datos

// Propiedades
data                      // Datos parseados
rawJson                   // JSON como string formateado
viewMode                  // 'tree' o 'raw'

// Eventos
@jsonviewer:copy          // Se dispara al copiar

    

  


  
  

    
HTMX + Django

    

      Visualiza respuestas JSON desde una API Django usando HTMX para carga dinamica.
    


    
      

      
Vista Django (views.py)

      
from django.http import JsonResponse
from django.views.decorators.http import require_GET

@require_GET
def api_user_detail(request, user_id):
    """Endpoint que retorna datos de usuario en JSON"""
    user = get_object_or_404(User, pk=user_id)
    return JsonResponse({
        'id': user.id,
        'username': user.username,
        'email': user.email,
        'is_active': user.is_active,
        'date_joined': user.date_joined.isoformat(),
        'profile': {
            'bio': user.profile.bio,
            'avatar': user.profile.avatar.url if user.profile.avatar else None,
            'preferences': user.profile.preferences
        }
    })

@require_GET
def api_orders_list(request):
    """Endpoint que retorna lista de pedidos"""
    orders = Order.objects.select_related('customer').all()[:10]
    return JsonResponse({
        'status': 200,
        'count': orders.count(),
        'results': [
            {
                'id': order.id,
                'customer': order.customer.username,
                'total': float(order.total),
                'status': order.status,
                'items': order.items.count(),
                'created_at': order.created_at.isoformat()
            }
            for order in orders
        ]
    })

Vista que retorna el fragment

      
import json
from django.shortcuts import render

def htmx_user_json(request, user_id):
    """Vista que retorna un fragment HTML con el JSON Viewer"""
    user = get_object_or_404(User, pk=user_id)
    user_data = {
        'id': user.id,
        'username': user.username,
        'email': user.email,
        'is_active': user.is_active,
        'profile': {
            'bio': user.profile.bio,
            'preferences': user.profile.preferences
        }
    }
    return render(request, 'partials/json_viewer.html', {
        'json_data': json.dumps(user_data),
        'title': f'Usuario #{user.id}',
        'badge': 'User'
    })

    


    
      

      
Uso con hx-on para procesar respuesta JSON directa

      
<!-- Cargar JSON directamente y procesarlo con Alpine.js -->
<div x-data="{
  jsonData: null,
  loading: false,
  async loadJson(url) {
    this.loading = true;
    try {
      const response = await fetch(url);
      this.jsonData = await response.json();
    } finally {
      this.loading = false;
    }
  }
}">
  <button class="ux-button ux-color-primary"
          @click="loadJson('{% url 'api_orders_list' %}')"
          :disabled="loading">
    <span x-show="loading" class="ux-spinner ux-spinner--sm"></span>
    <span x-text="loading ? 'Cargando...' : 'Cargar Pedidos'"></span>
  </button>

  <template x-if="jsonData">
    <div x-data="uxJsonViewer({ data: jsonData, initialExpanded: 3 })"
         class="ux-json-viewer" style="margin-top: 1rem;">
      <div class="ux-json-viewer__header">
        <div class="ux-json-viewer__title">
          <span>Pedidos</span>
          <span class="ux-json-viewer__badge"
                x-text="jsonData.count + ' items'"></span>
        </div>
        <div class="ux-json-viewer__actions">
          <button class="ux-json-viewer__btn" @click="expandAll()">Expandir</button>
          <button class="ux-json-viewer__btn" @click="collapseAll()">Colapsar</button>
          <button class="ux-json-viewer__btn" @click="copyJson()">Copiar</button>
        </div>
      </div>
      <div class="ux-json-viewer__content" style="max-height: 400px;">
        <pre class="ux-json-viewer__raw" x-text="rawJson">

# urls.py
from django.urls import path
from . import views

urlpatterns = [
    # API endpoints (retornan JSON)
    path('api/users//', views.api_user_detail, name='api_user_detail'),
    path('api/orders/', views.api_orders_list, name='api_orders_list'),

    # HTMX endpoints (retornan HTML fragments)
    path('htmx/users//json/', views.htmx_user_json, name='htmx_user_json'),
]
