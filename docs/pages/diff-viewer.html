HTMX + Django

Integra el diff viewer con Django para comparar versiones de documentos, historial de cambios o revisiones.

Vista Django

# views.py
from django.http import JsonResponse
from django.shortcuts import get_object_or_404
from .models import Document, DocumentVersion

def compare_versions(request, document_id):
    """Endpoint para comparar dos versiones de un documento."""
    document = get_object_or_404(Document, pk=document_id)

    old_version_id = request.GET.get('old')
    new_version_id = request.GET.get('new')

    old_version = get_object_or_404(DocumentVersion, pk=old_version_id, document=document)
    new_version = get_object_or_404(DocumentVersion, pk=new_version_id, document=document)

    return JsonResponse({
        'oldText': old_version.content,
        'newText': new_version.content,
        'oldTitle': f'Version {old_version.version_number}',
        'newTitle': f'Version {new_version.version_number}',
        'fileName': document.filename,
        'fileStatus': 'modified'
    })

def document_history(request, document_id):
    """Lista de versiones para selector."""
    document = get_object_or_404(Document, pk=document_id)
    versions = document.versions.order_by('-created_at')

    return render(request, 'partials/version_list.html', {
        'document': document,
        'versions': versions
    })

Modelo Django

# models.py
from django.db import models

class Document(models.Model):
    filename = models.CharField(max_length=255)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    def get_latest_version(self):
        return self.versions.order_by('-version_number').first()

class DocumentVersion(models.Model):
    document = models.ForeignKey(Document, on_delete=models.CASCADE, related_name='versions')
    version_number = models.PositiveIntegerField()
    content = models.TextField()
    author = models.ForeignKey('auth.User', on_delete=models.SET_NULL, null=True)
    created_at = models.DateTimeField(auto_now_add=True)
    commit_message = models.CharField(max_length=255, blank=True)

    class Meta:
        unique_together = ['document', 'version_number']
        ordering = ['-version_number']
