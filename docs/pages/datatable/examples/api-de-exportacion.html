// Metodos de exportacion disponibles
exportToCSV(options)     // Exportar a CSV
exportToExcel(options)   // Exportar a Excel (XML SpreadsheetML)
getExportData(selectedOnly)  // Obtener datos para exportar
quickExportCSV()         // Exportacion rápida CSV
quickExportExcel()       // Exportacion rápida Excel

// Opciones de exportToCSV
{
  filename: 'export',        // Nombre del archivo (sin extension)
  selectedOnly: false,       // Solo filas seleccionadas
  dateFormat: 'YYYY-MM-DD',  // Formato de fechas
  delimiter: ',',            // Delimitador (,  ;  \t)
  includeHeaders: true,      // Incluir cabeceras
  useRawValue: true          // Usar valor sin formato
}

// Opciones de exportToExcel
{
  filename: 'export',        // Nombre del archivo
  selectedOnly: false,       // Solo filas seleccionadas
  dateFormat: 'YYYY-MM-DD',  // Formato de fechas
  sheetName: 'Data',         // Nombre de la hoja
  useRawValue: true          // Usar valor sin formato
}

// Propiedades de columna para exportacion
{
  key: 'price',
  label: 'Precio',
  type: 'number',
  exportable: true,          // false para excluir de exportacion
  visible: true,             // false si columna esta oculta
  exportWidth: 100,          // Ancho en Excel
  exportFormat: (val) => val // Funcion de formato personalizado
}

// Propiedades reactivas de exportacion
exportMenuOpen              // Estado del menu desplegable
exportSelectedOnly          // Exportar solo seleccionados
exportVisibleColumnsOnly    // Solo columnas visibles

// Evento emitido
@export                     // { type: 'csv'|'excel', filename, rowCount }

// Formatos de fecha soportados
'YYYY-MM-DD'               // 2024-01-15
'DD/MM/YYYY'               // 15/01/2024
'MM/DD/YYYY'               // 01/15/2024
'YYYY-MM-DD HH:mm:ss'      // 2024-01-15 14:30:00
