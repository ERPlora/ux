Calendar Views

Vistas de calendario semanal y diaria con soporte para eventos.

Vista Semanal

Usa el componente uxCalendarView con view: 'week' para mostrar la semana completa.

Vista Diaria

Usa view: 'day' para mostrar solo un dia con todos sus eventos.

Uso

Formato de Evento

Los eventos requieren un objeto con las siguientes propiedades.

const event = {
  id: 1,                              // ID unico
  title: 'Reunion de equipo',         // Titulo del evento
  start: new Date('2026-01-10T09:00'), // Fecha/hora inicio
  end: new Date('2026-01-10T10:30'),   // Fecha/hora fin
  color: 'primary',                    // primary, success, danger, warning, secondary
  allDay: false                        // Evento de todo el dia
};

HTMX + Django

Integra el calendario con Django usando HTMX para cargar eventos dinamicamente desde el servidor.

Modelo Django

# models.py
from django.db import models

class Event(models.Model):
    COLOR_CHOICES = [
        ('primary', 'Primary'),
        ('success', 'Success'),
        ('warning', 'Warning'),
        ('danger', 'Danger'),
        ('secondary', 'Secondary'),
    ]

    title = models.CharField(max_length=200)
    start = models.DateTimeField()
    end = models.DateTimeField()
    color = models.CharField(max_length=20, choices=COLOR_CHOICES, default='primary')
    all_day = models.BooleanField(default=False)
    description = models.TextField(blank=True)
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ['start']

    def __str__(self):
        return self.title

Vista Django

# views.py
from django.http import JsonResponse
from django.views.decorators.http import require_GET, require_POST
from django.shortcuts import render
from datetime import datetime, timedelta
from .models import Event

@require_GET
def calendar_view(request):
    """Render the calendar page."""
    return render(request, 'calendar/index.html')

@require_GET
def get_events(request):
    """Get events for a date range (HTMX endpoint)."""
    start_date = request.GET.get('start')
    end_date = request.GET.get('end')

    if start_date and end_date:
        start = datetime.fromisoformat(start_date)
        end = datetime.fromisoformat(end_date)
        events = Event.objects.filter(start__gte=start, end__lte=end)
    else:
        # Default: current week
        today = datetime.now()
        start = today - timedelta(days=today.weekday())
        end = start + timedelta(days=7)
        events = Event.objects.filter(start__gte=start, end__lte=end)

    events_data = [{
        'id': e.id,
        'title': e.title,
        'start': e.start.isoformat(),
        'end': e.end.isoformat(),
        'color': e.color,
        'allDay': e.all_day,
    } for e in events]

    return JsonResponse({'events': events_data})

@require_POST
def create_event(request):
    """Create a new event (HTMX endpoint)."""
    import json
    data = json.loads(request.body)

    event = Event.objects.create(
        title=data['title'],
        start=datetime.fromisoformat(data['start']),
        end=datetime.fromisoformat(data['end']),
        color=data.get('color', 'primary'),
        all_day=data.get('allDay', False),
    )

    return JsonResponse({
        'id': event.id,
        'title': event.title,
        'start': event.start.isoformat(),
        'end': event.end.isoformat(),
        'color': event.color,
    })

@require_POST
def delete_event(request, event_id):
    """Delete an event (HTMX endpoint)."""
    Event.objects.filter(id=event_id).delete()
    return JsonResponse({'success': True})

URLs

# urls.py
from django.urls import path
from . import views

urlpatterns = [
    path('calendar/', views.calendar_view, name='calendar'),
    path('api/events/', views.get_events, name='get_events'),
    path('api/events/create/', views.create_event, name='create_event'),
    path('api/events//delete/', views.delete_event, name='delete_event'),
]
