Vista Django (views.py)

from django.shortcuts import render
from django.db.models import Count
from .models import Machine, MachineAlert

def machines_dashboard(request):
    machines = Machine.objects.select_related(
        'operator', 'current_job'
    ).prefetch_related(
        'alerts'
    ).all()

    context = {'machines': machines}

    if request.headers.get('HX-Request'):
        return render(request, 'partials/machine_cards.html', context)

    return render(request, 'machines_dashboard.html', context)

def status_summary(request):
    summary = Machine.objects.values('status').annotate(count=Count('id'))
    status_counts = {s['status']: s['count'] for s in summary}

    context = {
        'running': status_counts.get('running', 0),
        'idle': status_counts.get('idle', 0),
        'maintenance': status_counts.get('maintenance', 0),
        'offline': status_counts.get('offline', 0),
        'error': status_counts.get('error', 0),
    }

    return render(request, 'partials/machine_status_bar.html', context)

Modelo Django (models.py)

from django.db import models

class Machine(models.Model):
    STATUS_CHOICES = [
        ('running', 'Running'),
        ('idle', 'Idle'),
        ('maintenance', 'Maintenance'),
        ('offline', 'Offline'),
        ('error', 'Error'),
    ]

    name = models.CharField(max_length=100)
    code = models.CharField(max_length=20, unique=True)
    location = models.CharField(max_length=100)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='offline')
    speed = models.IntegerField(null=True, blank=True)
    temperature = models.DecimalField(max_digits=5, decimal_places=1, null=True)
    output = models.IntegerField(null=True, blank=True)
    uptime_percentage = models.DecimalField(max_digits=5, decimal_places=2, default=0)
    operator = models.ForeignKey('Operator', on_delete=models.SET_NULL, null=True, blank=True)
    current_job = models.ForeignKey('WorkOrder', on_delete=models.SET_NULL, null=True, blank=True)
    current_shift = models.CharField(max_length=20, default='Turno A')

class MachineAlert(models.Model):
    SEVERITY_CHOICES = [
        ('info', 'Info'),
        ('warning', 'Warning'),
        ('error', 'Error'),
    ]

    machine = models.ForeignKey(Machine, on_delete=models.CASCADE, related_name='alerts')
    message = models.CharField(max_length=255)
    severity = models.CharField(max_length=20, choices=SEVERITY_CHOICES, default='info')
    created_at = models.DateTimeField(auto_now_add=True)
    resolved = models.BooleanField(default=False)

class WorkOrder(models.Model):
    code = models.CharField(max_length=50)
    product = models.CharField(max_length=100)
    quantity = models.IntegerField()
    completed = models.IntegerField(default=0)

    @property
    def progress(self):
        if self.quantity == 0:
            return 0
        return int((self.completed / self.quantity) * 100)
