# views.py from django.shortcuts import render from django.http import HttpResponse from .forms import ContactForm def contact_form(request): if request.method == 'POST': form = ContactForm(request.POST) if form.is_valid(): contact = form.save() # Retornar el nuevo item para agregarlo a la lista return render(request, 'contacts/partials/contact_item.html', { 'contact': contact }) else: form = ContactForm() return render(request, 'contacts/partials/form.html', { 'form': form })
