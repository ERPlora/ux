def shipping_details(request, shipping_id):
    shipping = ShippingMethod.objects.get(id=shipping_id)
    request.session['shipping_method'] = shipping_id
    return render(request, 'checkout/partials/shipping_summary.html', {
        'shipping': shipping,
    })
