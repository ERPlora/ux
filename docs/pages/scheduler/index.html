Scheduler

A powerful booking and appointment scheduler with day, week, and month views, resource columns, and drag-to-create functionality.

Day View

Basic day view showing appointments for a single day. Click on a time slot to create a new booking, or click on an existing booking to edit it.

Week View with Resources

Display multiple resources (staff, rooms, equipment) as columns. Ideal for salon bookings, meeting room reservations, or equipment scheduling.

15-Minute Intervals

Use interval: 15 for finer granularity. The scheduler shows quarter-hour lines for more precise booking.

Drag on the time grid to create a new booking.

Glass Variant

Use the ux-scheduler--glass class for a frosted glass appearance following iOS design language.

Booking Format

Bookings require the following structure.

const booking = {
  id: 1,                              // Unique ID
  title: 'Team Meeting',              // Display title
  start: new Date('2026-01-10T09:00'), // Start date/time
  end: new Date('2026-01-10T10:30'),   // End date/time
  status: 'confirmed',                 // confirmed, pending, cancelled, blocked, primary, danger
  resourceId: 'room-1'                 // Optional: resource ID for multi-resource views
};

Django Model

# models.py
from django.db import models

class Resource(models.Model):
    name = models.CharField(max_length=100)
    info = models.CharField(max_length=200, blank=True)
    avatar = models.ImageField(upload_to='resources/', blank=True)
    is_active = models.BooleanField(default=True)

    def __str__(self):
        return self.name

class Booking(models.Model):
    STATUS_CHOICES = [
        ('confirmed', 'Confirmed'),
        ('pending', 'Pending'),
        ('cancelled', 'Cancelled'),
        ('blocked', 'Blocked'),
    ]

    resource = models.ForeignKey(Resource, on_delete=models.CASCADE, null=True, blank=True)
    title = models.CharField(max_length=200)
    start = models.DateTimeField()
    end = models.DateTimeField()
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='pending')
    notes = models.TextField(blank=True)
    customer_name = models.CharField(max_length=100, blank=True)
    customer_email = models.EmailField(blank=True)
    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ['start']

    def __str__(self):
        return f"{self.title} - {self.start.strftime('%Y-%m-%d %H:%M')}"

Django Views

# views.py
from django.http import JsonResponse
from django.shortcuts import render, get_object_or_404
from django.views.decorators.http import require_GET, require_POST
from datetime import datetime, timedelta
from .models import Booking, Resource
import json

@require_GET
def scheduler_view(request):
    """Render the scheduler page."""
    resources = Resource.objects.filter(is_active=True)
    return render(request, 'scheduler/index.html', {'resources': resources})

@require_GET
def get_bookings(request):
    """Get bookings for a date range (JSON API)."""
    start_date = request.GET.get('start')
    end_date = request.GET.get('end')
    resource_id = request.GET.get('resource_id')

    bookings = Booking.objects.all()

    if start_date:
        bookings = bookings.filter(start__gte=datetime.fromisoformat(start_date))
    if end_date:
        bookings = bookings.filter(end__lte=datetime.fromisoformat(end_date))
    if resource_id:
        bookings = bookings.filter(resource_id=resource_id)

    data = [{
        'id': b.id,
        'title': b.title,
        'start': b.start.isoformat(),
        'end': b.end.isoformat(),
        'status': b.status,
        'resourceId': b.resource_id,
    } for b in bookings]

    return JsonResponse({'bookings': data})

@require_POST
def create_booking(request):
    """Create a new booking (JSON API)."""
    data = json.loads(request.body)

    booking = Booking.objects.create(
        title=data['title'],
        start=datetime.fromisoformat(data['start']),
        end=datetime.fromisoformat(data['end']),
        status=data.get('status', 'pending'),
        resource_id=data.get('resourceId'),
        customer_name=data.get('customerName', ''),
        customer_email=data.get('customerEmail', ''),
        notes=data.get('notes', ''),
    )

    return JsonResponse({
        'id': booking.id,
        'title': booking.title,
        'start': booking.start.isoformat(),
        'end': booking.end.isoformat(),
        'status': booking.status,
        'resourceId': booking.resource_id,
    })

@require_POST
def update_booking(request, booking_id):
    """Update a booking."""
    booking = get_object_or_404(Booking, pk=booking_id)
    data = json.loads(request.body)

    if 'title' in data:
        booking.title = data['title']
    if 'start' in data:
        booking.start = datetime.fromisoformat(data['start'])
    if 'end' in data:
        booking.end = datetime.fromisoformat(data['end'])
    if 'status' in data:
        booking.status = data['status']
    if 'resourceId' in data:
        booking.resource_id = data['resourceId']

    booking.save()

    return JsonResponse({'success': True})

@require_POST
def delete_booking(request, booking_id):
    """Delete a booking."""
    Booking.objects.filter(id=booking_id).delete()
    return JsonResponse({'success': True})

<script>
// Extend component with API integration
document.addEventListener('alpine:init', () => {
  Alpine.data('uxScheduler', (options) => ({
    ...window.UX._components.uxScheduler(options),

    async loadBookings() {
      const start = this.getWeekStart(this.currentDate);
      const end = new Date(start);
      end.setDate(end.getDate() + 7);

      const response = await fetch(
        `/api/bookings/?start=${start.toISOString()}&end=${end.toISOString()}`
      );
      const data = await response.json();

      this.bookings = data.bookings.map(b => ({
        ...b,
        start: new Date(b.start),
        end: new Date(b.end)
      }));
    },

    async openBookingModal(detail) {
      // Open modal with booking form
      const modal = document.getElementById('booking-modal');
      modal.querySelector('[name="start"]').value = detail.start.toISOString().slice(0, 16);
      modal.querySelector('[name="end"]').value = detail.end?.toISOString().slice(0, 16) || '';
      modal.querySelector('[name="resourceId"]').value = detail.resourceId || '';
      modal.classList.add('ux-modal--open');
    },

    async saveBooking(formData) {
      const response = await fetch('/api/bookings/create/', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(Object.fromEntries(formData))
      });

      if (response.ok) {
        await this.loadBookings();
        document.getElementById('booking-modal').classList.remove('ux-modal--open');
      }
    }
  }));
});
</script>
