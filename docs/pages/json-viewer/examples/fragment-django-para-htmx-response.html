import json
from django.shortcuts import render

def htmx_user_json(request, user_id):
    """Vista que retorna un fragment HTML con el JSON Viewer"""
    user = get_object_or_404(User, pk=user_id)
    user_data = {
        'id': user.id,
        'username': user.username,
        'email': user.email,
        'is_active': user.is_active,
        'profile': {
            'bio': user.profile.bio,
            'preferences': user.profile.preferences
        }
    }
    return render(request, 'partials/json_viewer.html', {
        'json_data': json.dumps(user_data),
        'title': f'Usuario #{user.id}',
        'badge': 'User'
    })
