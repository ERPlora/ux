Django Views

# views.py
from django.shortcuts import render
from django.views.decorators.http import require_http_methods
from datetime import date, datetime, timedelta
from .models import AttendanceRecord, Employee

@require_http_methods(["GET"])
def attendance_list(request):
    """HTMX endpoint - returns attendance list partial"""
    selected_date = request.GET.get('date', date.today().isoformat())
    try:
        selected_date = datetime.strptime(selected_date, '%Y-%m-%d').date()
    except ValueError:
        selected_date = date.today()

    records = AttendanceRecord.objects.filter(
        date=selected_date
    ).select_related('employee').order_by('employee__name')

    return render(request, 'attendance/partials/attendance_list.html', {
        'attendance_records': records,
        'selected_date': selected_date
    })

@require_http_methods(["GET"])
def attendance_summary(request):
    """HTMX endpoint - returns summary stats"""
    selected_date = request.GET.get('date', date.today().isoformat())
    try:
        selected_date = datetime.strptime(selected_date, '%Y-%m-%d').date()
    except ValueError:
        selected_date = date.today()

    records = AttendanceRecord.objects.filter(date=selected_date)
    stats = {
        'total': Employee.objects.count(),
        'present': records.filter(status='present').count(),
        'absent': records.filter(status='absent').count(),
        'late': records.filter(status='late').count(),
        'leave': records.filter(status='leave').count(),
        'remote': records.filter(status='remote').count(),
    }

    return render(request, 'attendance/partials/summary.html', {'stats': stats})

@require_http_methods(["POST"])
def clock_in(request, employee_id):
    """HTMX endpoint - clock in employee"""
    record, created = AttendanceRecord.objects.get_or_create(
        employee_id=employee_id,
        date=date.today(),
        defaults={'status': 'present', 'clock_in': datetime.now().time()}
    )
    if not created and not record.clock_in:
        record.clock_in = datetime.now().time()
        record.status = 'late' if datetime.now().hour >= 9 else 'present'
        record.save()

    return render(request, 'attendance/partials/attendance_item.html', {
        'record': record
    })

Django Model

# models.py
from django.db import models
from datetime import datetime, timedelta

class Employee(models.Model):
    name = models.CharField(max_length=100)
    department = models.CharField(max_length=50)
    avatar = models.ImageField(upload_to='avatars/', blank=True, null=True)

    @property
    def initials(self):
        parts = self.name.split()
        if len(parts) >= 2:
            return f"{parts[0][0]}{parts[-1][0]}".upper()
        return self.name[:2].upper()

class AttendanceRecord(models.Model):
    STATUS_CHOICES = [
        ('present', 'Present'),
        ('absent', 'Absent'),
        ('late', 'Late'),
        ('leave', 'On Leave'),
        ('remote', 'Remote'),
    ]

    employee = models.ForeignKey(Employee, on_delete=models.CASCADE)
    date = models.DateField()
    status = models.CharField(max_length=10, choices=STATUS_CHOICES, default='absent')
    clock_in = models.TimeField(null=True, blank=True)
    clock_out = models.TimeField(null=True, blank=True)

    class Meta:
        unique_together = ['employee', 'date']

    @property
    def is_late(self):
        if self.clock_in:
            return self.clock_in.hour >= 9 or (self.clock_in.hour == 9 and self.clock_in.minute > 0)
        return False

    @property
    def work_hours(self):
        if self.clock_in and self.clock_out:
            delta = datetime.combine(self.date, self.clock_out) - datetime.combine(self.date, self.clock_in)
            return delta.total_seconds() / 3600
        return None

    @property
    def work_hours_display(self):
        hours = self.work_hours
        if hours:
            h = int(hours)
            m = int((hours - h) * 60)
            return f"{h}h {m}m"
        return None
