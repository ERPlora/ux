Django View

# views.py
from django.http import HttpResponse
from django.template.loader import render_to_string
from .models import Promotion

def active_promotion(request):
    """Retorna el banner de la promocion activa."""
    promo = Promotion.objects.filter(
        is_active=True,
        start_date__lte=timezone.now(),
        end_date__gte=timezone.now()
    ).first()

    if not promo:
        return HttpResponse('')

    # Verificar si el usuario ya lo descarto
    if request.user.is_authenticated:
        if promo.dismissed_by.filter(id=request.user.id).exists():
            return HttpResponse('')

    html = render_to_string('partials/promo_banner.html', {
        'promo': promo
    })
    return HttpResponse(html)

def dismiss_promotion(request, promo_id):
    """Marca la promocion como descartada para el usuario."""
    if request.user.is_authenticated:
        promo = Promotion.objects.get(id=promo_id)
        promo.dismissed_by.add(request.user)
    return HttpResponse(status=204)

def click_promotion(request, promo_id):
    """Registra un click en la promocion."""
    Promotion.objects.filter(id=promo_id).update(
        clicks=F('clicks') + 1
    )
    return HttpResponse(status=204)

Django Model

# models.py
from django.db import models
from django.contrib.auth import get_user_model

User = get_user_model()

class Promotion(models.Model):
    VARIANT_CHOICES = [
        ('info', 'Info'),
        ('success', 'Success'),
        ('warning', 'Warning'),
        ('danger', 'Danger'),
    ]

    title = models.CharField(max_length=100, blank=True)
    message = models.CharField(max_length=255)
    variant = models.CharField(max_length=20, choices=VARIANT_CHOICES, default='info')
    icon = models.TextField(blank=True, help_text='SVG icon HTML')
    action_url = models.URLField(blank=True)
    action_text = models.CharField(max_length=50, blank=True)

    is_active = models.BooleanField(default=True)
    start_date = models.DateTimeField()
    end_date = models.DateTimeField()

    clicks = models.PositiveIntegerField(default=0)
    dismissed_by = models.ManyToManyField(User, blank=True, related_name='dismissed_promos')

    created_at = models.DateTimeField(auto_now_add=True)

    class Meta:
        ordering = ['-created_at']

    def __str__(self):
        return self.title or self.message[:50]
