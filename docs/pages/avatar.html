Vista Django

# views.py
from django.shortcuts import render

def obtener_avatar(request, usuario_id):
    usuario = Usuario.objects.get(id=usuario_id)
    return render(request, 'partials/avatar_usuario.html', {
        'usuario': usuario
    })

def miembros_proyecto(request, proyecto_id):
    proyecto = Proyecto.objects.get(id=proyecto_id)
    miembros = proyecto.miembros.all()[:5]
    total = proyecto.miembros.count()

    return render(request, 'partials/avatar_grupo.html', {
        'miembros': miembros,
        'restantes': total - 5 if total > 5 else 0
    })
