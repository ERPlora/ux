from django.shortcuts import render
from django.http import JsonResponse
import json

def production_line_partial(request, line_id):
    line = ProductionLine.objects.get(id=line_id)
    stations = line.stations.all()

    context = {
        'line': {
            'name': line.name,
            'status': line.status,
            'daily_target': line.daily_target,
            'current_output': line.current_output,
            'efficiency': float(line.efficiency),
            'speed': line.speed,
            'downtime_minutes': line.calculate_downtime(),
            'stations_json': json.dumps([
                {
                    'name': s.name,
                    'status': s.status,
                    'output': s.output,
                    'alerts': s.alerts.count()
                }
                for s in stations
            ])
        }
    }
    return render(request, 'manufacturing/partials/production_line.html', context)
