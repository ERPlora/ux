Componente Alpine.js

Uso del componente uxRange con controles programaticos.

// Configuracion del componente uxRange
uxRange({
  value: 50,      // Valor inicial
  min: 0,         // Valor minimo
  max: 100,       // Valor maximo
  step: 5,        // Incremento
  disabled: false // Estado deshabilitado
})

// Metodos disponibles:
updateValue(newValue) // Establece un valor
increment(amount)     // Incrementa por step o cantidad
decrement(amount)     // Decrementa por step o cantidad
setMin()              // Establece al minimo
setMax()              // Establece al maximo
reset()               // Restaura al valor inicial

// Propiedades computadas:
percent      // Porcentaje actual (0-100)
thumbPosition // Posicion CSS del thumb

Componente uxDualRange

Para seleccionar rangos con dos valores.

// Configuracion del componente uxDualRange
uxDualRange({
  minValue: 20,  // Valor minimo seleccionado
  maxValue: 80,  // Valor maximo seleccionado
  min: 0,        // Limite inferior
  max: 100,      // Limite superior
  step: 1,       // Incremento
  minGap: 10,    // Separacion minima entre valores
  disabled: false
})

// Metodos:
updateMin(value) // Actualiza el valor minimo
updateMax(value) // Actualiza el valor maximo
reset()          // Restaura valores iniciales

// Propiedades computadas:
minPercent // Porcentaje del valor minimo
maxPercent // Porcentaje del valor maximo
fillStyle  // Estilos CSS para el fill { left, width }
range      // Array [minValue, maxValue]

Integracion con HTMX

Sliders que sincronizan con el servidor en tiempo real.

<!-- Guardar valor al soltar el slider -->
<div class="ux-range" x-data="uxRange({ value: 50 })">
  <span class="ux-range__label">Volumen</span>
  <div class="ux-range__wrapper">
    <div class="ux-range__track-container">
      <div class="ux-range__track">
        <div class="ux-range__fill" :style="{ width: percent + '%' }"></div>
      </div>
      <input type="range" class="ux-range__input"
             name="volume"
             min="0" max="100"
             x-model="value"
             hx-post="/api/settings/volume"
             hx-trigger="change"
             hx-swap="none">
      <div class="ux-range__thumb" :style="{ left: percent + '%' }"></div>
    </div>
  </div>
</div>
<!-- Filtro de rango de precios con dual range -->
<form hx-get="/api/products"
      hx-trigger="change from:find input"
      hx-target="#product-list">
  <div class="ux-range ux-range--dual"
       x-data="uxDualRange({ minValue: 0, maxValue: 1000 })">
    <span class="ux-range__label">Rango de precio</span>
    <div class="ux-range__wrapper">
      <span class="ux-range__value" x-text="'$' + minValue"></span>
      <div class="ux-range__track-container">
        <div class="ux-range__track">
          <div class="ux-range__fill" :style="fillStyle"></div>
        </div>
        <input type="range" name="price_min"
               min="0" max="1000" x-model="minValue"
               @input="updateMin($event.target.value)">
        <input type="range" name="price_max"
               min="0" max="1000" x-model="maxValue"
               @input="updateMax($event.target.value)">
      </div>
      <span class="ux-range__value" x-text="'$' + maxValue"></span>
    </div>
  </div>
</form>
