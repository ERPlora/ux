Validacion en tiempo real

Mostrar errores de validacion mientras el usuario escribe.

El email ya esta registrado en el sistema.




  
  



# Django view
from django.http import HttpResponse
from django.template.loader import render_to_string

def validate_email(request):
    email = request.POST.get('email', '')

    if User.objects.filter(email=email).exists():
        return HttpResponse(render_to_string(
            'partials/callout_error.html',
            {'message': 'El email ya esta registrado en el sistema.'}
        ))

    return HttpResponse('')  # Sin errores

# Django view
def system_notices(request):
    notices = SystemNotice.objects.filter(
        active=True,
        start_date__lte=timezone.now(),
        end_date__gte=timezone.now()
    )

    return render(request, 'partials/system_notices.html', {
        'notices': notices
    })
