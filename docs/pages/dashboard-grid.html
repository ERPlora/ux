Dashboard Interactivo con Alpine.js

Dashboard con drag-and-drop para reordenar widgets. Haz clic en "Editar" para activar el modo de edicion.

Mi Dashboard 
<div x-data="uxDashboardGrid({
  items: [
    { id: 'users', title: 'Usuarios', size: 'sm' },
    { id: 'revenue', title: 'Ingresos', size: 'md' },
    { id: 'chart', title: 'Grafico', size: 'full' }
  ],
  columns: 4,
  editing: false
})">
  <button @click="toggleEdit()">Editar</button>

  <div class="ux-dashboard-grid" :class="{ 'ux-dashboard-grid--editing': editing }">
    <template x-for="(item, index) in items" :key="item.id">
      <div class="ux-dashboard-grid__item"
           :class="getItemSizeClass(item.size)"
           :draggable="editing"
           @dragstart="onDragStart(index, $event)"
           @dragover="onDragOver(index, $event)"
           @drop="onDrop(index, $event)">
        <!-- Widget content -->
      </div>
    </template>
  </div>
</div>

API de Alpine.js

Metodos y propiedades disponibles en el componente uxDashboardGrid().

Propiedades

// Configuracion inicial
uxDashboardGrid({
  items: [],           // Array de widgets
  editing: false,      // Modo de edicion activo
  columns: 4,          // Numero de columnas (2, 3, 4, 6)
  gap: 'md'            // Espaciado: 'none', 'sm', 'md', 'lg'
})

// Estructura de un item
{
  id: 'widget-1',      // ID unico
  title: 'Mi Widget',  // Titulo
  size: 'md',          // Tamano: 'sm', 'md', 'lg', 'full'
  // ...otros datos personalizados
}

Metodos

// Edicion
toggleEdit()                  // Alternar modo de edicion
startEdit()                   // Activar modo de edicion
endEdit()                     // Desactivar modo de edicion

// CRUD de items
addItem({ title, size })      // Agregar widget
removeItem(id)                // Eliminar widget
updateItem(id, { ... })       // Actualizar widget
resizeItem(id, size)          // Cambiar tamano

// Ordenamiento
moveItem(fromIndex, toIndex)  // Mover widget de posicion

// Persistencia
serialize()                   // Obtener estado serializable
restore(savedItems)           // Restaurar desde estado guardado

Eventos

// Escuchar eventos
@dashboard:add="..."       // Widget agregado
@dashboard:remove="..."    // Widget eliminado
@dashboard:update="..."    // Widget actualizado
@dashboard:move="..."      // Widget movido
@dashboard:resize="..."    // Widget redimensionado
@dashboard:editchange="..."// Modo edicion cambiado
@dashboard:dragstart="..." // Inicio de drag
@dashboard:dragend="..."   // Fin de drag

# views.py
from django.http import JsonResponse
from django.shortcuts import render
from django.views.decorators.http import require_POST
from django.contrib.auth.decorators import login_required
import json

@login_required
def dashboard_view(request):
    """Render dashboard page with user's widget layout."""
    user_layout = DashboardLayout.objects.filter(user=request.user).first()
    widgets = user_layout.widgets if user_layout else get_default_widgets()
    return render(request, 'dashboard/index.html', {
        'widgets_json': json.dumps(widgets)
    })

@login_required
def widgets_list(request):
    """Return all dashboard widgets as HTML partial."""
    user_layout = DashboardLayout.objects.filter(user=request.user).first()
    widgets = user_layout.widgets if user_layout else get_default_widgets()
    return render(request, 'dashboard/partials/widgets.html', {
        'widgets': widgets
    })

@login_required
def widget_content(request, widget_id=None):
    """Return content for a specific widget."""
    widget_id = widget_id or request.GET.get('id')
    widget_data = get_widget_data(widget_id, request.user)
    return render(request, f'dashboard/widgets/{widget_id}.html', {
        'data': widget_data
    })

@login_required
def widget_refresh(request, widget_id):
    """Refresh widget data and return updated content."""
    widget_data = get_widget_data(widget_id, request.user, refresh=True)
    return render(request, f'dashboard/widgets/{widget_id}.html', {
        'data': widget_data
    })

@login_required
@require_POST
def save_layout(request):
    """Save user's dashboard layout."""
    layout_data = json.loads(request.POST.get('layout', '[]'))
    layout, created = DashboardLayout.objects.update_or_create(
        user=request.user,
        defaults={'widgets': layout_data}
    )
    return JsonResponse({'success': True})

def get_widget_data(widget_id, user, refresh=False):
    """Get data for a specific widget."""
    widget_handlers = {
        'sales': lambda u: {
            'total': Order.objects.filter(user=u).aggregate(Sum('total'))['total__sum'] or 0,
            'count': Order.objects.filter(user=u).count()
        },
        'users': lambda u: {
            'active': User.objects.filter(is_active=True).count(),
            'new_today': User.objects.filter(date_joined__date=date.today()).count()
        },
        'orders': lambda u: {
            'pending': Order.objects.filter(status='pending').count(),
            'recent': list(Order.objects.order_by('-created')[:5].values())
        },
    }
    handler = widget_handlers.get(widget_id, lambda u: {})
    return handler(user)

def get_default_widgets():
    """Return default widget configuration."""
    return [
        {'id': 'users', 'title': 'Usuarios', 'size': 'sm'},
        {'id': 'sales', 'title': 'Ingresos', 'size': 'sm'},
        {'id': 'orders', 'title': 'Pedidos', 'size': 'sm'},
        {'id': 'chart', 'title': 'Ventas Mensuales', 'size': 'full'},
    ]

# urls.py
from django.urls import path
from . import views

app_name = 'dashboard'
urlpatterns = [
    path('', views.dashboard_view, name='index'),
    path('widgets/', views.widgets_list, name='widgets'),
    path('widget//', views.widget_content, name='widget-content'),
    path('widget//refresh/', views.widget_refresh, name='widget-refresh'),
    path('layout/save/', views.save_layout, name='save-layout'),
]
