Alpine.js Component API

The uxLeaveRequestForm component provides state management for the form.

// Configuration options
x-data="uxLeaveRequestForm({
  // Initial values
  leaveType: 'vacation',
  startDate: '2024-01-15',
  endDate: '2024-01-19',
  reason: '',

  // Custom leave types
  leaveTypes: [
    { id: 'vacation', name: 'Vacation', icon: 'sun' },
    { id: 'sick', name: 'Sick', icon: 'medical' },
    // ...
  ],

  // Leave balance (for validation)
  balance: {
    vacation: 15,
    sick: 10,
    personal: 3
  },

  // Reset form after submit
  resetOnSubmit: true
})"

// Available properties
leaveType       // Selected leave type ID
startDate       // Start date (YYYY-MM-DD)
endDate         // End date (YYYY-MM-DD)
reason          // Reason text
isHalfDay       // Boolean for half-day requests
duration        // Computed working days
currentBalance  // Balance for selected type
remainingBalance // Balance after request
isBalanceLow    // True if remaining < 3
isValid         // Form validation state
errors          // Validation error messages
submitting      // Submit in progress

// Available methods
selectType(typeId)  // Select leave type
validate()          // Validate form
submit()            // Submit request (emits 'leave-request:submit')
reset()             // Reset form
cancel()            // Cancel and reset (emits 'leave-request:cancel')

// Events
@leave-request:submit="handleSubmit($event.detail)"
@leave-request:cancel="handleCancel()"

# models.py
class LeaveRequest(models.Model):
    LEAVE_TYPES = [
        ('vacation', 'Vacation'),
        ('sick', 'Sick Leave'),
        ('personal', 'Personal'),
        ('maternity', 'Maternity'),
        ('paternity', 'Paternity'),
        ('bereavement', 'Bereavement'),
        ('unpaid', 'Unpaid Leave'),
        ('work-from-home', 'Work From Home'),
    ]

    STATUS_CHOICES = [
        ('pending', 'Pending'),
        ('approved', 'Approved'),
        ('rejected', 'Rejected'),
        ('cancelled', 'Cancelled'),
    ]

    employee = models.ForeignKey(Employee, on_delete=models.CASCADE)
    leave_type = models.CharField(max_length=20, choices=LEAVE_TYPES)
    start_date = models.DateField()
    end_date = models.DateField()
    reason = models.TextField(blank=True)
    status = models.CharField(max_length=20, choices=STATUS_CHOICES, default='pending')
    created_at = models.DateTimeField(auto_now_add=True)
    approved_by = models.ForeignKey(User, null=True, blank=True, on_delete=models.SET_NULL)

    @property
    def duration(self):
        """Calculate working days between dates."""
        days = 0
        current = self.start_date
        while current <= self.end_date:
            if current.weekday() < 5:  # Monday=0 to Friday=4
                days += 1
            current += timedelta(days=1)
        return days

@require_POST
def approve_leave(request, pk):
    leave_request = get_object_or_404(LeaveRequest, pk=pk)
    leave_request.status = 'approved'
    leave_request.approved_by = request.user
    leave_request.save()
    return render(request, 'partials/leave_request_card.html', {
        'request': leave_request
    })
