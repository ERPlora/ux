Vista Django

# views.py
from django.http import HttpResponse

def contar_notificaciones(request):
    count = Notificacion.objects.filter(
        usuario=request.user,
        leida=False
    ).count()

    # Si es 0, retornar vacio para ocultar badge
    if count == 0:
        return HttpResponse('')

    return HttpResponse(str(count))
