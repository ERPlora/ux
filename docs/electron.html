// main.js (Electron main process)
const { autoUpdater } = require('electron-updater');

// Configurar auto-updater
autoUpdater.autoDownload = false;
autoUpdater.autoInstallOnAppQuit = true;

// Verificar actualizaciones al iniciar
app.whenReady().then(() => {
  autoUpdater.checkForUpdates();
});

// Eventos
autoUpdater.on('update-available', (info) => {
  mainWindow.webContents.send('update-available', info);
});

autoUpdater.on('update-downloaded', (info) => {
  mainWindow.webContents.send('update-downloaded', info);
});

// IPC handlers
ipcMain.handle('download-update', () => {
  autoUpdater.downloadUpdate();
});

ipcMain.handle('quit-and-install', () => {
  autoUpdater.quitAndInstall();
});

Preload Script

Configuracion del preload script para exponer APIs seguras al renderer.

// preload.js
const { contextBridge, ipcRenderer } = require('electron');

contextBridge.exposeInMainWorld('electronAPI', {
  // Window controls
  minimize: () => ipcRenderer.invoke('window-minimize'),
  maximize: () => ipcRenderer.invoke('window-maximize'),
  unmaximize: () => ipcRenderer.invoke('window-unmaximize'),
  close: () => ipcRenderer.invoke('window-close'),
  setFullscreen: (flag) => ipcRenderer.invoke('window-fullscreen', flag),

  // Window state
  onMaximize: (callback) => ipcRenderer.on('window-maximized', callback),
  onUnmaximize: (callback) => ipcRenderer.on('window-unmaximized', callback),
  onFocus: (callback) => ipcRenderer.on('window-focus', callback),
  onBlur: (callback) => ipcRenderer.on('window-blur', callback),

  // App info
  getVersion: () => ipcRenderer.invoke('get-version'),
  getPath: (name) => ipcRenderer.invoke('get-path', name),

  // Shell
  openExternal: (url) => ipcRenderer.invoke('open-external', url),
  showItemInFolder: (path) => ipcRenderer.invoke('show-item-in-folder', path),

  // Dialogs
  showOpenDialog: (options) => ipcRenderer.invoke('show-open-dialog', options),
  showSaveDialog: (options) => ipcRenderer.invoke('show-save-dialog', options),
  showMessageBox: (options) => ipcRenderer.invoke('show-message-box', options),

  // Notifications
  showNotification: (options) => ipcRenderer.invoke('show-notification', options),

  // Auto-update
  checkForUpdates: () => ipcRenderer.invoke('check-for-updates'),
  downloadUpdate: () => ipcRenderer.invoke('download-update'),
  quitAndInstall: () => ipcRenderer.invoke('quit-and-install'),
  onUpdateAvailable: (callback) => ipcRenderer.on('update-available', (_, info) => callback(info)),
  onUpdateDownloaded: (callback) => ipcRenderer.on('update-downloaded', (_, info) => callback(info)),

  // Store (electron-store)
  storeGet: (key) => ipcRenderer.invoke('store-get', key),
  storeSet: (key, value) => ipcRenderer.invoke('store-set', key, value),
  storeDelete: (key) => ipcRenderer.invoke('store-delete', key),

  // Generic IPC
  send: (channel, ...args) => ipcRenderer.send(channel, ...args),
  invoke: (channel, ...args) => ipcRenderer.invoke(channel, ...args)
});

Main Process Handlers

Handlers IPC en el proceso principal de Electron.

// main.js
const { app, BrowserWindow, ipcMain, shell, dialog, Notification } = require('electron');
const path = require('path');
const Store = require('electron-store');

const store = new Store();
let mainWindow;

function createWindow() {
  mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    frame: false, // Frameless para custom titlebar
    titleBarStyle: 'hidden', // macOS: ocultar pero mantener traffic lights
    trafficLightPosition: { x: 16, y: 10 }, // Posicion traffic lights
    webPreferences: {
      preload: path.join(__dirname, 'preload.js'),
      contextIsolation: true,
      nodeIntegration: false
    }
  });

  mainWindow.loadFile('index.html');

  // Window state events
  mainWindow.on('maximize', () => {
    mainWindow.webContents.send('window-maximized');
  });

  mainWindow.on('unmaximize', () => {
    mainWindow.webContents.send('window-unmaximized');
  });

  mainWindow.on('focus', () => {
    mainWindow.webContents.send('window-focus');
  });

  mainWindow.on('blur', () => {
    mainWindow.webContents.send('window-blur');
  });
}

// Window control handlers
ipcMain.handle('window-minimize', () => mainWindow.minimize());
ipcMain.handle('window-maximize', () => mainWindow.maximize());
ipcMain.handle('window-unmaximize', () => mainWindow.unmaximize());
ipcMain.handle('window-close', () => mainWindow.close());
ipcMain.handle('window-fullscreen', (_, flag) => mainWindow.setFullScreen(flag));

// App info handlers
ipcMain.handle('get-version', () => app.getVersion());
ipcMain.handle('get-path', (_, name) => app.getPath(name));

// Shell handlers
ipcMain.handle('open-external', (_, url) => shell.openExternal(url));
ipcMain.handle('show-item-in-folder', (_, path) => shell.showItemInFolder(path));

// Dialog handlers
ipcMain.handle('show-open-dialog', (_, options) => dialog.showOpenDialog(mainWindow, options));
ipcMain.handle('show-save-dialog', (_, options) => dialog.showSaveDialog(mainWindow, options));
ipcMain.handle('show-message-box', (_, options) => dialog.showMessageBox(mainWindow, options));

// Notification handler
ipcMain.handle('show-notification', (_, options) => {
  new Notification(options).show();
});

// Store handlers
ipcMain.handle('store-get', (_, key) => store.get(key));
ipcMain.handle('store-set', (_, key, value) => store.set(key, value));
ipcMain.handle('store-delete', (_, key) => store.delete(key));

app.whenReady().then(createWindow);

Deep Links (Protocol Handler)

Manejar URLs personalizadas como myapp://action.

// main.js - Registrar protocolo
if (process.defaultApp) {
  if (process.argv.length >= 2) {
    app.setAsDefaultProtocolClient('myapp', process.execPath, [path.resolve(process.argv[1])]);
  }
} else {
  app.setAsDefaultProtocolClient('myapp');
}

// Manejar deep link en macOS
app.on('open-url', (event, url) => {
  event.preventDefault();
  handleDeepLink(url);
});

// Manejar deep link en Windows/Linux (single instance)
const gotTheLock = app.requestSingleInstanceLock();

if (!gotTheLock) {
  app.quit();
} else {
  app.on('second-instance', (event, commandLine) => {
    // Windows: el deep link viene en commandLine
    const url = commandLine.find(arg => arg.startsWith('myapp://'));
    if (url) handleDeepLink(url);

    // Enfocar ventana
    if (mainWindow) {
      if (mainWindow.isMinimized()) mainWindow.restore();
      mainWindow.focus();
    }
  });
}

function handleDeepLink(url) {
  const parsed = new URL(url);
  console.log('Deep link:', parsed.pathname, parsed.searchParams);

  // Enviar al renderer
  mainWindow.webContents.send('deep-link', {
    path: parsed.pathname,
    params: Object.fromEntries(parsed.searchParams)
  });
}
// preload.js
contextBridge.exposeInMainWorld('electronAPI', {
  // ... otros handlers
  onDeepLink: (callback) => ipcRenderer.on('deep-link', (_, data) => callback(data))
});

// renderer.js
window.electronAPI.onDeepLink((data) => {
  console.log('Deep link received:', data);
  // Navegar a la ruta
  if (data.path === '/order') {
    navigateTo(`/orders/${data.params.id}`);
  }
});

API Reference

UXElectron Methods

Method Description
minimize() Minimize window
maximize() Maximize window
unmaximize() Restore window from maximized
toggleMaximize() Toggle maximize state
close() Close window
setFullscreen(flag) Enter/exit fullscreen
openExternal(url) Open URL in default browser
showItemInFolder(path) Show file in file manager
showOpenDialog(options) Show native open file dialog
showSaveDialog(options) Show native save file dialog
showMessageBox(options) Show native message box
showNotification(title, body) Show native notification
checkForUpdates() Check for app updates
downloadUpdate() Download available update
installUpdate() Install update and restart
storeGet(key) Get value from persistent store
storeSet(key, value) Set value in persistent store
invoke(channel, ...args) Send async IPC message

Alpine Component Properties

Property Type Description
isElectron boolean Running in Electron
platform string darwin, win32, or linux
isMaximized boolean Window maximized state
isFocused boolean Window focus state
updateAvailable boolean Update available
updateDownloaded boolean Update downloaded
updateVersion string Available update version
