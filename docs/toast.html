Toast Basico

Notificaciones temporales en la parte inferior.

// Con Alpine.js
<div x-data="uxToastManager()">
  <button @click="show('Mensaje del toast')">Mostrar</button>

  <div class="ux-toast-container">
    <template x-for="toast in toasts">
      <div class="ux-toast" x-show="toast.visible">
        <span x-text="toast.message"></span>
      </div>
    </template>
  </div>
</div>

</div>

API Alpine - Toast Manager

Componente completo para gestionar multiples toasts.

<div x-data="uxToastManager()">
  <!-- Botones para mostrar toasts -->
  <button @click="show('Mensaje')">Default</button>
  <button @click="success('Guardado!')">Success</button>
  <button @click="error('Error!')">Error</button>
  <button @click="warning('Cuidado!')">Warning</button>
  <button @click="info('Info!')">Info</button>

  <!-- Contenedor de toasts -->
  <div class="ux-toast-container">
    <template x-for="toast in toasts" :key="toast.id">
      <div class="ux-toast"
           :class="'ux-toast--' + toast.type"
           x-show="toast.visible"
           x-transition>
        <span x-text="toast.message"></span>
        <button @click="dismiss(toast.id)">X</button>
      </div>
    </template>
  </div>
</div>

Integracion con HTMX

HTMX es la tecnologia principal para mostrar toasts desde respuestas del servidor.

Toast desde respuesta de formulario

<!-- Formulario que muestra toast al guardar -->
<form hx-post="/api/settings"
      hx-target="#toast-container"
      hx-swap="beforeend">
  <input type="text" name="name" class="ux-input">
  <button type="submit" class="ux-button ux-color-primary">Guardar</button>
</form>

<!-- Contenedor fijo para toasts -->
<div id="toast-container" class="ux-toast-container ux-toast-container--bottom"></div>

<!-- Servidor devuelve toast con auto-cierre -->
<div class="ux-toast ux-toast--success ux-toast--visible"
     x-data="{ show: true }"
     x-show="show"
     x-init="setTimeout(() => { show = false; setTimeout(() => $el.remove(), 300) }, 3000)">
  <span class="ux-toast__message">Configuracion guardada</span>
</div>

Toast con accion de reintentar

<!-- Boton que puede fallar -->
<button id="sync-btn"
        hx-post="/api/sync"
        hx-target="#toast-container"
        hx-swap="beforeend">
  Sincronizar
</button>

<!-- Servidor devuelve toast de error con accion -->
<div class="ux-toast ux-toast--danger ux-toast--visible">
  <span class="ux-toast__message">Error de conexion</span>
  <button class="ux-toast__action"
          hx-post="/api/sync"
          hx-target="#toast-container"
          hx-swap="beforeend">
    Reintentar
  </button>
</div>

Toast con Out-of-Band (OOB)

<!-- Formulario normal -->
<form hx-post="/api/products"
      hx-target="#products-table"
      hx-swap="beforeend">
  <input type="text" name="product" class="ux-input">
  <button type="submit" class="ux-button ux-color-primary">Agregar</button>
</form>

<div id="toast-container" class="ux-toast-container"></div>

<!-- Servidor devuelve fila + toast OOB -->
<tr>
  <td>Nuevo Producto</td>
  <td>$99.00</td>
</tr>

<!-- Toast inyectado via OOB -->
<div id="toast-container" hx-swap-oob="beforeend">
  <div class="ux-toast ux-toast--success ux-toast--visible" ...>
    Producto agregado
  </div>
</div>

Toast para delete con confirmacion

        hx-target="#item-123"
        hx-swap="outerHTML"
        hx-confirm="Eliminar este item?">
  Eliminar
</button>

<!-- Servidor devuelve espacio vacio + toast OOB -->
<div></div>

<div id="toast-container" hx-swap-oob="beforeend">
  <div class="ux-toast ux-toast--success ux-toast--visible"
       x-data="{ show: true }"
       x-show="show"
       x-init="setTimeout(() => { show = false; setTimeout(() => $el.remove(), 300) }, 3000)">
    <span class="ux-toast__message">Item eliminado</span>

            hx-post="/api/items/123/restore"
            hx-target="#items-list"
            hx-swap="beforeend">
      Deshacer
    </button>
  </div>
</div>

API Alpine

Metodo Descripcion
show(message, options) Muestra un toast con mensaje y opciones
success(message) Toast de exito
error(message) Toast de error
warning(message) Toast de advertencia
info(message) Toast informativo
dismiss(id) Cierra un toast especifico
dismissAll() Cierra todos los toasts
setPosition(pos) Cambia la posicion del contenedor
containerClass Clase CSS dinamica para el contenedor

Posiciones disponibles

top, bottom, center, top-start, top-end, bottom-start, bottom-end, center-start, center-end
